// ---------------------------------------------
// 3. EXPANSI√ìN SEM√ÅNTICA ‚Äî MULTILENGUAJE + IA + MANUAL + NORMALIZACI√ìN
// ---------------------------------------------
function expandQuery(rawQuery) {
    const exploded = explodeStrings(rawQuery);
    const normalized = exploded.map((t) => normalize(t));
    const expanded = new Set();

    normalized.forEach((token) => {
        if (!token) return;

        expanded.add(token);

        // IA DICTIONARY
        if (aiDictionary[token]) {
            aiDictionary[token].forEach((s) => expanded.add(normalize(s)));
        }

        // MANUAL SYNONYMS
        if (manualSynonyms[token]) {
            manualSynonyms[token].forEach((s) => expanded.add(normalize(s)));
        }

        // Tok√©n sem√°ntico adicional:
        if (token.endsWith("ia")) expanded.add(token.replace("ia", "y"));
    });

    return [...expanded];
}

// ---------------------------------------------
// 4. BUILD FUSE INDEX ‚Äî NORMALIZADO + OPTIMIZADO
// ---------------------------------------------
export function buildFuseIndex(firms) {
    const indexed = firms.map((f) => ({
        ...f,
        nameNorm: normalize(f.name || ""),
        countryNorm: normalize(f.country || ""),
        areaNorm: normalize(f.area || ""),
        descriptionNorm: normalize(f.description || "").slice(0, 300), // üî• Optimizaci√≥n #3
        rankedNorm: normalize(String(f.ranked || "")),
        tagsNormalized: (f.tags || []).map((t) => normalize(t)), // üî• FIX importante
    }));

    const options = {
        includeScore: true,
        includeMatches: true,
        ignoreLocation: true,
        threshold: 0.38,
        distance: 120, // üî• Optimizaci√≥n #2
        minMatchCharLength: 2,
        keys: [
            { name: "nameNorm", weight: 0.45 },
            { name: "areaNorm", weight: 0.35 },
            { name: "countryNorm", weight: 0.25 },
            { name: "descriptionNorm", weight: 0.15 },
            { name: "tagsNormalized", weight: 0.25 },
        ],
    };

    console.log("Fuse index built with", indexed.length, "records");
    return new Fuse(indexed, options);
}

// ---------------------------------------------
// 5. GENERADOR DE EXPLICACI√ìN DETALLADA
// ---------------------------------------------
function generateExplanation(item, matches, matchedTerms) {
    const fields = matches.map((m) => m.key).join(", ");
    const trimmed = matchedTerms.slice(0, 5).join(", ");

    return (
        `Campo(s) coincidente(s): ${fields}. ` +
        `T√©rminos relevantes: ${trimmed}.`
    );
}

// ---------------------------------------------
// 6. SEMANTIC SEARCH ULTRA-FAST ‚Äî MULTI-TERM OR MATCH
// ---------------------------------------------
export function semanticSearch(fuse, query, limit = 20) {
    if (!query || !query.trim()) return [];

    const normQuery = normalize(query);

    // üî• OPTIMIZACI√ìN #4 ‚Äî CACHE
    if (CACHE.has(normQuery)) return CACHE.get(normQuery);

    // 1. Expansi√≥n de t√©rminos
    const expandedTerms = expandQuery(query);

    // 2. B√öSQUEDA POR T√âRMINO (OR logic)
    const combined = [];
    const seen = new Set();

    for (const term of expandedTerms) {
        if (!term || term.length < 2) continue;

        const results = fuse.search(term).slice(0, 50); // üî• Optimizaci√≥n #1

        for (const r of results) {
            const id = r.item.id;
            if (!seen.has(id)) {
                seen.add(id);
                combined.push({ term, ...r });
            }
        }
    }

    // 3. Armado de resultados con explicaci√≥n
    const final = combined
        .map((entry) => {
            const matchedTerms = [entry.term];
            return {
                id: entry.item.id,
                name: entry.item.name,
                country: entry.item.country,
                area: entry.item.area,
                ranked: entry.item.ranked,
                description: entry.item.description,
                tags: entry.item.tags,
                score: entry.score,
                matchedTerms,
                explanation: generateExplanation(entry.item, entry.matches || [], matchedTerms),
            };
        })
        .sort((a, b) => a.score - b.score)
        .slice(0, limit);

    // 4. Cachear y devolver
    CACHE.set(normQuery, final);
    return final;
}

// ---------------------------------------------
// 7. B√öSQUEDA POR TAG (NORMALIZADA)
// ---------------------------------------------
export function searchByTag(firms, tag) {
    const target = normalize(tag);

    return firms.filter((f) =>
        f.tagsNormalized?.includes(target)
    );
}

// ---------------------------------------------
// 8. TAGS R√ÅPIDOS
// ---------------------------------------------
export function getQuickTags(firms) {
    const tagCounter = {};
    firms.forEach((f) => {
        (f.tags || []).forEach((t) => {
            const norm = normalize(t);
            tagCounter[norm] = (tagCounter[norm] || 0) + 1;
        });
    });
    return Object.entries(tagCounter)
        .map(([tag, count]) => ({ tag, count }))
        .sort((a, b) => b.count - a.count);
}

// ---------------------------------------------
// 9. ESTAD√çSTICAS
// ---------------------------------------------
export function getIndexStats() {
    return {
        aiSynonyms: Object.keys(aiDictionary).length,
        manualSynonyms: Object.keys(manualSynonyms).length,
        cacheEntries: CACHE.size,
    };
}

// ---------------- PDF generator ----------------
function generatePDF(payload) {
    return new Promise((resolve) => {
        const LOGO_PATH = path.join(process.cwd(), "logo.svg");
        const filePath = path.join(TEMP_DIR, `reporte_${Date.now()}.pdf`);

        const doc = new PDFDocument({
            size: "A4",
            margins: { top: 40, bottom: 40, left: 50, right: 50 }
        });

        const stream = fs.createWriteStream(filePath);
        doc.pipe(stream);

        // ===========================================================
        // ENCABEZADO
        // ===========================================================
        const headerHeight = 90;

        doc.save();
        doc.rect(0, 0, doc.page.width, headerHeight).fill("#0f172a");
        doc.restore();

        try {
            doc.image(LOGO_PATH, 50, 22, { width: 50 });
        } catch (e) { }

        doc.fillColor("#ffffff")
            .font("Helvetica-Bold")
            .fontSize(22)
            .text("Reporte del Formulario", 120, 30);

        doc.moveDown(3);

        // ===========================================================
        // T√çTULO BLOQUE
        // ===========================================================
        const drawBlock = (title) => {
            doc.moveDown(1.2);
            doc.fillColor("#0f172a")
                .font("Helvetica-Bold")
                .fontSize(15)
                .text(title);

            doc.moveDown(0.4);

            doc.strokeColor("#d7d7d7")
                .lineWidth(1)
                .moveTo(50, doc.y)
                .lineTo(doc.page.width - 50, doc.y)
                .stroke();

            doc.moveDown(1);
        };

        // ===========================================================
        // TABLA REAL ‚Äî INFORMACI√ìN DEL ENCARGO
        // ===========================================================

        drawBlock("Informaci√≥n del Encargo");

        const tableData = [
            ["Nombre", payload.name],
            ["Correo", payload.email],
            ["Pa√≠s", payload.country],
            ["√Årea", payload.area],
            ["A√±os en el mercado", payload.marketYears],
            ["Experiencia del abogado a cargo", payload.leadExperience],
            ["Tipo de firma", Array.isArray(payload.firmType) ? payload.firmType.join(", ") : payload.firmType],
            ["Otras caracter√≠sticas", Array.isArray(payload.otherCharacteristics) ? payload.otherCharacteristics.join(" ¬∑ ") : payload.otherCharacteristics],
            ["Urgencia", payload.urgency],
            ["Preferencia de cobro", payload.billingPreference],
            ["Encargo", payload.assignment],
            ["Comentarios", payload.comments],
        ];

        const colLabelX = 50;
        const colLabelW = 170;

        const colValueX = colLabelX + colLabelW;
        const colValueW = doc.page.width - colValueX - 50;

        const rowHeight = 22;
        let yTable = doc.y;

        // Header (solo estilo visual, misma tabla)
        doc.rect(colLabelX, yTable, colLabelW, rowHeight)
            .fill("#0f172a");
        doc.rect(colValueX, yTable, colValueW, rowHeight)
            .fill("#0f172a");

        doc.fillColor("#ffffff")
            .font("Helvetica-Bold")
            .fontSize(11)
            .text("DATOS", colLabelX + 6, yTable + 6)
            .text("", colValueX + 6, yTable + 6);

        yTable += rowHeight;

        // Rows
        tableData.forEach((row, i) => {
            const bg = i % 2 === 0 ? "#f5f5f5" : "#ffffff";

            // Label cell
            doc.rect(colLabelX, yTable, colLabelW, rowHeight).fill(bg);
            doc.fillColor("#0f172a")
                .font("Helvetica-Bold")
                .fontSize(10)
                .text(row[0], colLabelX + 6, yTable + 6);

            // Value cell
            doc.rect(colValueX, yTable, colValueW, rowHeight).fill(bg);
            doc.fillColor("#333333")
                .font("Helvetica")
                .fontSize(10)
                .text(row[1] || "No proporcionado", colValueX + 6, yTable + 6, {
                    width: colValueW - 12
                });

            yTable += rowHeight;

            // Salto de p√°gina si es necesario
            if (yTable > doc.page.height - 80) {
                doc.addPage();
                yTable = 70;
            }
        });

        doc.moveDown(2);

        // ===========================================================
        // BLOQUE ‚Äî FIRMAS RECOMENDADAS (NO SE EDITA)
        // ===========================================================
        drawBlock("Firmas Recomendadas");

        if (!Array.isArray(payload.savedFirms) || !payload.savedFirms.length) {
            doc.font("Helvetica")
                .fontSize(12)
                .fillColor("#444")
                .text("No hay firmas recomendadas.");

            doc.end();
            return resolve(filePath);
        }

        // Tabla original (sin cambios)
        const tableTop = doc.y;
        const col1 = 50;
        const col1w = 150;
        const col2 = col1 + col1w + 10;
        const col2w = 100;
        const col3 = col2 + col2w + 10;
        const col3w = doc.page.width - col3 - 50;

        const rowHeight2 = 22;


        doc.rect(col1, tableTop, col1w, rowHeight2).fill("#0f172a");
        doc.rect(col2, tableTop, col2w, rowHeight2).fill("#0f172a");
        doc.rect(col3, tableTop, col3w, rowHeight2).fill("#0f172a");

        doc.fillColor("#ffffff")
            .font("Helvetica-Bold")
            .fontSize(12);

        doc.text("Firma", col1 + 6, tableTop + 6);

function escapeHtml(text = "") {
    return String(text)
        .replace(/&/g, "&amp;")
        .replace(/</g, "&lt;")
        .replace(/>/g, "&gt;");
}

// Trunca un texto a N caracteres y agrega "..."
function truncateText(text = "", maxLength = 90) {
    const str = String(text || "").trim();
    if (!str) return "";
    if (str.length <= maxLength) return str;
    return str.slice(0, maxLength) + "...";
}


function buildHTMLEmail(payload = {}) {

        (v === undefined || v === null || v === "")

        Array.isArray(a) && a.length
            ? a.map(x => `<li>${escapeHtml(String(x))}</li>`).join("")
            : "<li><em>No aplica</em></li>";

    // --- Agrupar firmas por pa√≠s ---
    const savedFirms = Array.isArray(payload.savedFirms) ? payload.savedFirms : [];
    const countryMap = new Map();

        const country = (f.country || "Sin pa√≠s").trim();
        if (!countryMap.has(country)) {
            countryMap.set(country, { country, firmNames: [], areas: [] });
        }
        const entry = countryMap.get(country);
        if (f.firm) entry.firmNames.push(f.firm);
        if (f.area) entry.areas.push(f.area);

    const groupedRowsHTML = savedFirms.length
        ? Array.from(countryMap.values())

                return `

              <td style="border:1px solid #ccc; padding:6px;">

              </td>
            </tr>
          `;
            })
            .join("")
        : "";

    return `

          <tr><td style="padding:8px; background:#f7f7f7; font-weight:600;">Tipo de firma</td><td style="padding:8px;">${Array.isArray(payload.firmType) ? escapeHtml(payload.firmType.join(', ')) : safe(payload.firmType)}</td></tr>

// ---------------- PDF generator ----------------
function generatePDF(payload) {
    return new Promise((resolve) => {
        const LOGO_PATH = path.join(process.cwd(), "logo.svg");
        const filePath = path.join(TEMP_DIR, `reporte_${Date.now()}.pdf`);

        const doc = new PDFDocument({
            size: "A4",
            margins: { top: 40, bottom: 40, left: 50, right: 50 }
        });

        const stream = fs.createWriteStream(filePath);
        doc.pipe(stream);

        // ===========================================================
        // ENCABEZADO
        // ===========================================================
        const headerHeight = 90;

        doc.save();
        doc.rect(0, 0, doc.page.width, headerHeight).fill("#0f172a");
        doc.restore();

        try {
            doc.image(LOGO_PATH, 50, 22, { width: 50 });
        } catch (e) { }

        doc.fillColor("#ffffff")
            .font("Helvetica-Bold")
            .fontSize(22)
            .text("Reporte del Formulario", 120, 30);

        doc.moveDown(3);

        // ===========================================================
        // T√çTULO BLOQUE
        // ===========================================================
        const drawBlock = (title) => {
            doc.moveDown(1.2);
            doc.fillColor("#0f172a")
                .font("Helvetica-Bold")
                .fontSize(15)
                .text(title);

            doc.moveDown(0.4);

            doc.strokeColor("#d7d7d7")
                .lineWidth(1)
                .moveTo(50, doc.y)
                .lineTo(doc.page.width - 50, doc.y)
                .stroke();

            doc.moveDown(1);
        };

        // ===========================================================
        // TABLA ‚Äî INFORMACI√ìN DEL ENCARGO
        // ===========================================================
        drawBlock("Informaci√≥n del Encargo");

        const tableData = [
            ["Nombre", payload.name],
            ["Correo", payload.email],
            ["Pa√≠s", payload.country],
            ["√Årea", payload.area],
            ["A√±os en el mercado", payload.marketYears],
            ["Experiencia del abogado a cargo", payload.leadExperience],
            ["Tipo de firma", Array.isArray(payload.firmType) ? payload.firmType.join(", ") : payload.firmType],
            ["Otras caracter√≠sticas", Array.isArray(payload.otherCharacteristics) ? payload.otherCharacteristics.join(" ¬∑ ") : payload.otherCharacteristics],
            ["Urgencia", payload.urgency],
            ["Preferencia de cobro", payload.billingPreference],
            ["Encargo", payload.assignment],
            ["Comentarios", payload.comments],
        ];

        const colLabelX = 50;
        const colLabelW = 170;

        const colValueX = colLabelX + colLabelW;
        const colValueW = doc.page.width - colValueX - 50;

        const rowHeight = 22;
        let yTable = doc.y;

        // Header
        doc.rect(colLabelX, yTable, colLabelW, rowHeight)
            .fill("#0f172a");
        doc.rect(colValueX, yTable, colValueW, rowHeight)
            .fill("#0f172a");

        doc.fillColor("#ffffff")
            .font("Helvetica-Bold")
            .fontSize(11)
            .text("DATOS", colLabelX + 6, yTable + 6);

        yTable += rowHeight;

        // Filas
        tableData.forEach((row, i) => {
            const bg = i % 2 === 0 ? "#f5f5f5" : "#ffffff";

            doc.rect(colLabelX, yTable, colLabelW, rowHeight).fill(bg);
            doc.fillColor("#0f172a")
                .font("Helvetica-Bold")
                .fontSize(10)
                .text(row[0], colLabelX + 6, yTable + 6);

            doc.rect(colValueX, yTable, colValueW, rowHeight).fill(bg);
            doc.fillColor("#333333")
                .font("Helvetica")
                .fontSize(10)
                .text(row[1] || "No proporcionado", colValueX + 6, yTable + 6, {
                    width: colValueW - 12
                });

            yTable += rowHeight;

            if (yTable > doc.page.height - 80) {
                doc.addPage();
                yTable = 70;
            }
        });

        doc.moveDown(2);

        // ===========================================================
        // BLOQUE ‚Äî FIRMAS RECOMENDADAS (AGRUPADAS POR PA√çS)
        // ===========================================================
        drawBlock("Firmas Recomendadas");

        const savedFirms = Array.isArray(payload.savedFirms) ? payload.savedFirms : [];
        if (!savedFirms.length) {
            doc.font("Helvetica")
                .fontSize(12)
                .fillColor("#444")
                .text("No hay firmas recomendadas.");

            doc.end();
            return resolve(filePath);
        }

        // Agrupar por pa√≠s
        const countryMap = new Map();
        savedFirms.forEach(f => {
            const country = (f.country || "Sin pa√≠s").trim();
            if (!countryMap.has(country)) {
                countryMap.set(country, { country, firmNames: [], areas: [] });
            }
            const entry = countryMap.get(country);
            if (f.firm) entry.firmNames.push(f.firm);
            if (f.area) entry.areas.push(f.area);
        });

        const groupedCountries = Array.from(countryMap.values());

        const tableTop = doc.y;
        const colCountry = 50;
        const colCountryW = 80;
        const colFirm = colCountry + colCountryW + 10;
        const colFirmW = 180;
        const colAreas = colFirm + colFirmW + 10;
        const colAreasW = doc.page.width - colAreas - 50;

        const headerH = 22;

        // Encabezado tabla
        doc.rect(colCountry, tableTop, colCountryW, headerH).fill("#0f172a");
        doc.rect(colFirm, tableTop, colFirmW, headerH).fill("#0f172a");
        doc.rect(colAreas, tableTop, colAreasW, headerH).fill("#0f172a");

        doc.fillColor("#ffffff")
            .font("Helvetica-Bold")
            .fontSize(11);

        doc.text("Pa√≠s", colCountry + 6, tableTop + 5);
        doc.text("Firma(s)", colFirm + 6, tableTop + 5);
        doc.text("√Åreas", colAreas + 6, tableTop + 5);

        let y = tableTop + headerH;

        groupedCountries.forEach((c, idx) => {
            const bg = idx % 2 === 0 ? "#f5f5f5" : "#ffffff";
            const firmsText = c.firmNames.join(" ¬∑ ");
            const areasJoined = c.areas.join(" | ");
            const areasTruncated = truncateText(areasJoined, 90); // üëà truncamos aqu√≠

            const rowH = 40; // un poco m√°s alto para permitir wrapping

            doc.rect(colCountry, y, colCountryW, rowH).fill(bg);
            doc.rect(colFirm, y, colFirmW, rowH).fill(bg);
            doc.rect(colAreas, y, colAreasW, rowH).fill(bg);

            doc.fillColor("#333")
                .font("Helvetica")
                .fontSize(10);

            doc.text(c.country, colCountry + 6, y + 6, { width: colCountryW - 12 });
            doc.text(firmsText, colFirm + 6, y + 6, { width: colFirmW - 12 });
            doc.text(areasTruncated, colAreas + 6, y + 6, { width: colAreasW - 12 });

            y += rowH;

            if (y > doc.page.height - 80) {
                doc.addPage();
                y = 70;
            }
        });

        doc.moveDown(2);

        doc.fontSize(10)
            .fillColor("#777")
            .text("Fecha de generaci√≥n: " + new Date().toLocaleString());

        doc.end();

        stream.on("finish", () => resolve(filePath));
    });
}

// üîπ Endpoint para recibir el formulario y guardar en Google Sheets
app.post("/api/form-submit", async (req, res) => {
    try {
        const payload = req.body || {};

        // Asegurarnos de que savedFirms sea siempre un arreglo
        const savedFirms = Array.isArray(payload.savedFirms) ? payload.savedFirms : [];

        const dataToSave = {
            ...payload,
            savedFirms,
            // asegurar los nuevos campos (si vienen vac√≠os que sean string vac√≠o)
            marketYears: payload.marketYears || "",
            leadExperience: payload.leadExperience || ""
        };

        // 1) Guardar en Google Sheets (tu funci√≥n existente)
        await appendFormResponse(dataToSave);

        // 2) Intentar enviar correo (si est√° configurado)
        if (canSendEmail()) {
            try {
                const mailOptions = {
                    from: `"${FROM_NAME || "Formulario"}" <${FROM_EMAIL || SMTP_USER}>`,
                    to: TO_EMAIL,
                    subject: `Nuevo env√≠o formulario ‚Äî ${payload.name || payload.email || 'Sin nombre'}`,
                    html: buildHTMLEmail(dataToSave),
                    text: `Nuevo env√≠o de formulario. Nombre: ${payload.name || ''} - Email: ${payload.email || ''}`
                };

                // Generar PDF y adjuntar
                let pdfPath;
                try {
                    pdfPath = await generatePDF(dataToSave);
                    // adjuntar el pdf
                    mailOptions.attachments = [
                        {
                            filename: "reporte_formulario.pdf",
                            path: pdfPath
                        }
                    ];
                } catch (errPdf) {
                    console.error("‚ùå Error generando PDF:", errPdf);
                    // seguimos sin adjunto si falla
                }

                await transporter.sendMail(mailOptions);
                console.log("‚úîÔ∏è Email enviado a", TO_EMAIL);

                // limpiar pdf temporal (si fue generado)
                if (pdfPath) {
                    setTimeout(() => {
                        try { fs.unlinkSync(pdfPath); } catch (e) { /* ignore */ }
                    }, 2000);
                }
            } catch (errMail) {
                console.error("‚ùå Error enviando email:", errMail);
                // No interrumpimos la respuesta; devolvemos ok: true aunque falle el mail.
            }
        } else {
            console.warn("‚ö†Ô∏è Email no enviado: configuraci√≥n SMTP/TO_EMAIL incompleta.");
        }

        return res.json({ ok: true });
    } catch (err) {
        console.error("‚ùå Error en /api/form-submit:", err);
        return res.status(500).json({ ok: false, error: "Error al guardar en Google Sheets" });
    }
});

/**
 * POST /api/form-pdf
 * Body: payload (mismo formato que se env√≠a en /api/form-submit)
 * Response: application/pdf (stream)
 */
app.post("/api/form-pdf", async (req, res) => {
    try {
        const payload = req.body || {};
        // asegurar arrays/strings
        payload.savedFirms = Array.isArray(payload.savedFirms) ? payload.savedFirms : [];
        payload.marketYears = payload.marketYears || "";
        payload.leadExperience = payload.leadExperience || "";

        const pdfPath = await generatePDF(payload);

        // Stream the file to response
        res.setHeader('Content-Type', 'application/pdf');
        res.setHeader('Content-Disposition', `attachment; filename="reporte_formulario.pdf"`);

        const readStream = fs.createReadStream(pdfPath);
        readStream.pipe(res);

        // delete file after streaming finishes
        readStream.on('close', () => {
            try { fs.unlinkSync(pdfPath); } catch (e) { /* ignore */ }
        });

    } catch (err) {
        console.error("‚ùå Error en /api/form-pdf:", err);
        res.status(500).json({ ok: false, error: "Error generando PDF" });
    }
});
